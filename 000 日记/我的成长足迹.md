```dataviewjs
const calendarData = {
    year: 2026,  // 显示哪一年的数据
    colors: {    // 这里的颜色模仿 GitHub 的绿色
        green: ["#ebedf0", "#c6e48b", "#7bc96f", "#239a3b", "#196127"]
    },
    entries: []
}

// 自动扫描库里所有笔记，根据 created 属性进行统计
for (let page of dv.pages()) {
    if (page.created) {
        calendarData.entries.push({
            date: moment(page.created.toString()).format("YYYY-MM-DD"),
            intensity: 1, // 只要当天写了笔记，格子就变绿
            content: await dv.span(`[[${page.file.name}| ]]`)
        })
    }
}

renderHeatmapCalendar(this.container, calendarData)
```

```dataviewjs
// --- 在这里修改你想要的年份和月份 ---
const targetYear = 2026; 
const targetMonth = 3;   
// --------------------------------

const calendarData = {
    year: targetYear,
    colors: {
        green: ["#ebedf0", "#c6e48b", "#7bc96f", "#239a3b", "#196127"]
    },
    entries: []
}

for (let page of dv.pages()) {
    if (page.created) {
        const m = moment(page.created.toString());
        // 只有年份和月份都对上的笔记，才会被画在图上
        if (m.year() === targetYear && (m.month() + 1) === targetMonth) {
            calendarData.entries.push({
                date: m.format("YYYY-MM-DD"),
                intensity: 1,
                content: await dv.span(`[[${page.file.name}| ]]`)
            })
        }
    }
}

renderHeatmapCalendar(this.container, calendarData)
```
